# Test instantiation and inheritance
my_list = MyList([1, 2, 3])
print(isinstance(my_list, list))  # Should print: True

# Test print_sorted() with unsorted append
my_list.append(0)
my_list.print_sorted()  # Should print: [0, 1, 2, 3]
print(my_list)          # Should print: [1, 2, 3, 0]

# Test with negative numbers
my_list = MyList([3, -1, 2])
my_list.print_sorted()  # Should print: [-1, 2, 3]

# Test with empty list
my_list = MyList()
my_list.print_sorted()  # Should print: []
